select "Firstname", "Lastname" from "Doctor";

select "Firstname", "Lastname" from "Patient"
where "HospitalId" = 1

select p."Firstname" as "PatientFirstname", p."Lastname" as "PatientLastname", d."Firstname", d."Lastname", s."Description" 
from "Patient" p
inner join "DoctorPatient" dp
on dp."PatientId" = p."Id"
inner join "Doctor" d 
on dp."DoctorId" = d."Id" 
inner join "Specialization" s
on s."Id" = d."SpecializationId"
where s."Description" like 'Cardiology'

select h."Name", d."Firstname", d."Lastname"
from "Hospital" h
inner join "DoctorHospital" dh
on dh."HospitalId" = h."Id"
inner join "Doctor" d
on dh."DoctorId" = d."Id"

select d."Firstname" as "DoctorFirstname", d."Lastname" as "DoctorLastname", p."Firstname" as "PatientFirstname", p."Lastname" as "PatientLastname" 
from "Doctor" d
inner join "DoctorPatient" dp
on d."Id" = dp."DoctorId"
left join "Patient" p 
on p."Id" = dp."PatientId"

select d."Firstname" as "DoctorName", p."Firstname" as "PatientName", dp."Condition" as "Condition"
from "Doctor" d
inner join "DoctorPatient" dp
on dp."DoctorId" = d."Id"
inner join "Patient" p
on p."Id" = dp."PatientId"
where d."Firstname"  = 'Ana'
